Conditional Rendering Patterns

React provides several ways to conditionally render content. Let's explore the most common patterns:

1. If/Else with Variables

function WelcomeMessage({ user, isLoggedIn }) { let content; if (isLoggedIn) { content = <h1>Welcome back, {user.name}!</h1>; } else { content = <h1>Please sign in to continue.</h1>; } return ( <div> {content} <p>Thank you for visiting our site.</p> </div> ); }

2. Ternary Operator

function StatusMessage({ isOnline }) { return ( <div> <h2>User Status</h2> <p> You are currently {isOnline ? 'online' : 'offline'} {isOnline ? ' üü¢' : ' üî¥'} </p> {/* Nested ternary (use sparingly) */} <button> {isOnline ? 'Go Offline' : 'Connect' } </button> </div> ); }

3. Logical AND Operator

function NotificationBanner({ notifications, showBanner }) { return ( <div> {/* Show banner only if showBanner is true */} {showBanner && ( <div style={{ background: '#f0f8ff', padding: '10px' }}> <h3>Notifications</h3> </div> )} {/* Show count only if there are notifications */} {notifications.length > 0 && ( <p>You have {notifications.length} new notifications</p> )} {/* Show message only if no notifications */} {notifications.length === 0 && ( <p>No new notifications</p> )} </div> ); }

4. Switch-like Pattern with Objects

function StatusIcon({ status }) { const statusConfig = { loading: { icon: '‚è≥', color: '#ffc107', text: 'Loading...' }, success: { icon: '‚úÖ', color: '#28a745', text: 'Success!' }, error: { icon: '‚ùå', color: '#dc3545', text: 'Error occurred' }, warning: { icon: '‚ö†Ô∏è', color: '#fd7e14', text: 'Warning' } }; const config = statusConfig[status] || statusConfig.loading; return ( <div style={{ color: config.color }}> <span>{config.icon}</span> <span>{config.text}</span> </div> ); }

Rendering Lists

The map() function is the standard way to render lists in React:

function ProductList({ products }) { return ( <div> <h2>Our Products</h2> {products.length === 0 ? The Importance of Keys

Keys help React identify which items have changed, been added, or removed:

üîë Key Rules:
  • Keys must be unique among siblings
  • Keys should be stable (don't use array index if list can change)
  • Use meaningful identifiers when possible
function TodoList({ todos, onToggle, onDelete }) { return ( <ul> {todos.map(todo => ( <li key={todo.id} // ‚úÖ Good - stable, unique ID style={{ textDecoration: todo.completed ?     Filtering and Sorting Lists

    function FilterableProductList({ products }) { const [searchTerm, setSearchTerm] = useState(''); const [sortBy, setSortBy] = useState('name'); const [filterCategory, setFilterCategory] = useState('all'); const filteredAndSortedProducts = products .filter(product => { const matchesSearch = product.name .toLowerCase() .includes(searchTerm.toLowerCase()); const matchesCategory = filterCategory === 'all' || product.category === filterCategory; return matchesSearch && matchesCategory; }) .sort((a, b) => { if (sortBy === 'name') { return a.name.localeCompare(b.name); } if (sortBy === 'price') { return a.price - b.price; } return 0; }); const categories = [...new Set(products.map(p => p.category))]; return ( <div> <div className="filters"> <input type="text" placeholder="Search products..." value={searchTerm} onChange={(e) => setSearchTerm(e.target.value)} /> <select value={sortBy} onChange={(e) => setSortBy(e.target.value)} > <option value="name">Sort by Name</option> <option value="price">Sort by Price</option> </select> <select value={filterCategory} onChange={(e) => setFilterCategory(e.target.value)} > <option value="all">All Categories</option> {categories.map(category => ( <option key={category} value={category}> {category} </option> ))} </select> </div> <div className="results"> <p>Showing {filteredAndSortedProducts.length} products</p> {filteredAndSortedProducts.length === 0 ? ( <div className="no-results"> <p>No products match your criteria.</p> <button onClick={() => { setSearchTerm(''); setFilterCategory('all'); }}> Clear Filters </button> </div> ) : ( <div className="product-list"> {filteredAndSortedProducts.map(product => ( <div key={product.id} className="product-item"> <h3>{product.name}</h3> <p>Category: {product.category}</p> <p>Price: ${product.price}</p> </div> ))} </div> )} </div> </div> ); }
